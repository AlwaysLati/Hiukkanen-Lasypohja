\song{Macarena (\textit{431})}{}

            Annatko mulle pillua vai revaa, \\
            annatko sitä mikä viidakossa venaa. \\
            Annatko mulle pillua vai revaa, \\
            hei – anna revaa! \\
            Annatko mulle kullia vai melaa, \\
            annatko sitä mikä viidakossa venaa. \\
            Annatko mulle kullia vai melaa, \\
            hei – anna melaa! \\
