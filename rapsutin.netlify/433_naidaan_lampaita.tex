\song{Naidaan lampaita (\textit{433})}{Lambada}

            :,: Naidaan lampaita, lampailla ei ole hampaita. :,: \\
            :,: Laita lammas rinteeseen, pistä sorkat saappaaseen, \\
            työnnä sormi lampaan perseeseen. :,: \\
\hspace{10mm} \\
            :,: Naidaan lehmiä, lehmät eivät ehdi empiä. :,: \\
            :,: Laita kulli sieraimeen, niin sen kieli lipaisee, \\
            ihanasti sinun kivekseen. :,: \\
\hspace{10mm} \\
            :,: Naidaan norsuja, norsut eivät käytä kortsuja. :,: \\
            :,: Ota pesti sirkukseen, liity ihmeperheeseen, \\
            työnnä jalka norsun perseeseen. :,: \\
\hspace{10mm} \\
            :,: Naidaan poroja, porot ovat metsän horoja. :,: \\
            :,: Tartu kiinni sarvista, ota haju arvesta, \\
            vedä varvi koko tokasta. :,: \\
\hspace{10mm} \\
            :,: Naidaan majavaa, majavalta löytyy parrua. :,: \\
            :,: Ota kiinni hampaista, vedä parru padosta, \\
            työnnä parru pikku majavaan. :,: \\
\hspace{10mm} \\
            \begin{comment} Lisäsäkeistöt: \\
\hspace{10mm} \\
            :,: Naidaan lääkäärii, lääkäri tietää sinun tautisi. :,: \\
            :,: Ota pistos hauikseen, laita lääkäri kontilleen, \\
            työnnä neula lääkärin perseeseen. :,: \\
\hspace{10mm} \\
            :,: Naidaan fukseja, fukseilla ei ole tauteja. :,: \\
            :,: Vie fuksi kiltikseen, heitä sohvalle vattalleen, \\
            työnnä kyrpä fuksin perseeseen. :,: \\
\hspace{10mm} \\
            :,: Naidaan mäyriä, mäyrien vehkeet ovat käyriä. :,: \\
            :,: Vedä mutka suoraksi, ota mäyrä huoraksi, \\
            silloin saat sä kupan varmasti. :,: \\
\hspace{10mm} \\
            :,: Naidaan proffia, proffat antaa meille noppia. :,: \\
            :,: Vedä proffaa kakkoseen, niin se antaa nelosen, \\
            vedä kovaa niin antaa vitosen. :,: \\
\end{comment}
